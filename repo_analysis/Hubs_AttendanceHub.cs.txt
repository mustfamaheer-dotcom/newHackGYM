using System; using System.Threading.Tasks; using Microsoft.AspNet.SignalR; using Microsoft.AspNet.SignalR.Hubs; namespace AttendanceSystem.Hubs { /// /// SignalR hub that pushes real-time attendance events to connected /// dashboard clients. The Python device service POSTs records to the /// /api/attendance/sync endpoint, which then broadcasts to all clients /// subscribed to this hub. /// /// Client-side events (JavaScript): /// - attendanceRecorded : fired on each individual check-in/out /// - recordsSynced : fired when a batch sync completes /// - deviceStatusChanged : fired when device status updates /// public class AttendanceHub : Hub { private static int _connectedClients = 0; public override Task OnConnected() { _connectedClients++; Clients.All.clientCountChanged(_connectedClients); return base.OnConnected(); } public override Task OnDisconnected(bool stopCalled) { if (_connectedClients > 0) _connectedClients--; Clients.All.clientCountChanged(_connectedClients); return base.OnDisconnected(stopCalled); } /// /// Allows clients to join a device-specific group so they only /// receive events for a particular device serial number. /// [HubMethodName("SubscribeToDevice")] public Task SubscribeToDevice(string deviceSerial) { if (string.IsNullOrWhiteSpace(deviceSerial)) throw new ArgumentException("Device serial required"); return Groups.Add(Context.ConnectionId, $"device-{deviceSerial}"); } [HubMethodName("UnsubscribeFromDevice")] public Task UnsubscribeFromDevice(string deviceSerial) => string.IsNullOrWhiteSpace(deviceSerial) ? Task.CompletedTask : Groups.Remove(Context.ConnectionId, $"device-{deviceSerial}"); } }
