using System; using System.Linq; using System.Net; using System.Threading.Tasks; using System.Web.Http; using System.Web.Http.Cors; using AttendanceSystem.DTOs; using AttendanceSystem.Models; using AttendanceSystem.Services; namespace AttendanceSystem.Controllers { /// /// Handles attendance events: real-time recording, device sync, queries, reports. /// [RoutePrefix("api/attendance")] [EnableCors(origins: "*", headers: "*", methods: "*")] public class AttendanceController : ApiController { private readonly IAttendanceService _attendanceService; private readonly IUserService _userService; private readonly IReportService _reportService; public AttendanceController(IAttendanceService attendanceService, IUserService userService, IReportService reportService) { _attendanceService = attendanceService; _userService = userService; _reportService = reportService; } /// /// POST: /api/attendance/sync — called by the Python ZKTeco service /// to push batch attendance records from the device. /// [HttpPost] [Route("sync")] public async Task SyncAttendanceRecords([FromBody] SyncRequest request) { if (request?.Records == null || request.Records.Count == 0) return BadRequest("No records provided"); try { var inputs = request.Records.Select(r => new SyncRecordInput { UserId = r.UserId, Timestamp = r.Timestamp, Status = r.Status, DeviceSerialNumber = r.DeviceSerialNumber ?? request.DeviceSerialNumber, VerificationType = r.VerificationType, VerificationScore = r.VerificationScore, CapturedImage = r.CapturedImage }).ToList(); var result = await _attendanceService.SyncDeviceRecordsAsync(inputs, request.DeviceSerialNumber); return Ok(new ApiResponse { Success = true, Message = $"Synced {result.Inserted} new records ({result.Skipped} duplicates skipped)", Data = new { inserted = result.Inserted, skipped = result.Skipped, total = result.Total, deviceSerial = request.DeviceSerialNumber } }); } catch (Exception ex) { var msg = ex.Message; var inner = ex.InnerException; while (inner != null) { msg += " | " + inner.GetType().Name + ": " + inner.Message; inner = inner.InnerException; } System.Diagnostics.Trace.TraceError("Sync failed: " + ex); return InternalServerError(new Exception(msg, ex)); } } /// /// POST: /api/attendance/checkin — record a single attendance event. /// [HttpPost] [Route("checkin")] public async Task RecordCheckin([FromBody] CheckinRequest request) { if (request == null || request.UserId == 0) return BadRequest("Invalid request: UserId is required"); try { if (!Enum.TryParse(request.Status, true, out var status)) return BadRequest($"Invalid status '{request.Status}'. Use 'CheckIn' or 'CheckOut'"); var verification = request.VerificationType.HasValue ? (VerificationType)request.VerificationType.Value : VerificationType.FaceRecognition; var score = request.VerificationScore ?? 0; var record = await _attendanceService.RecordAttendanceAsync( request.UserId, status, request.CapturedImage, verification, score); return Ok(new ApiResponse { Success = true, Message = "Attendance recorded successfully", Data = new { id = record.Id, userId = record.UserId, timestamp = record.Timestamp, status = record.Status.ToString(), verificationType = record.VerificationType.ToString() } }); } catch (InvalidOperationException ex) { return Content(HttpStatusCode.BadRequest, new ApiResponse { Success = false, Message = ex.Message }); } catch (Exception ex) { return InternalServerError(ex); } } /// /// GET: /api/attendance/today — recent attendance events for live dashboard. /// [HttpGet] [Route("today")] public async Task GetTodayAttendance() { try { var records = await _attendanceService.GetTodayAttendanceAsync(); var users = await _userService.GetAllUsersAsync(); var userMap = users.ToDictionary(u => u.UserId); var dtos = records .OrderByDescending(r => r.Timestamp) .Select(r => new { id = r.Id, userId = r.UserId, employeeId = userMap.TryGetValue(r.UserId, out var u) ? u.EmployeeId : null, userName = userMap.TryGetValue(r.UserId, out u) ? u.Name : null, department = userMap.TryGetValue(r.UserId, out u) ? u.Department : null, timestamp = r.Timestamp, status = r.Status.ToString(), verificationType = r.VerificationType.ToString(), verificationScore = r.VerificationScore, deviceSerialNumber = r.DeviceSerialNumber, hasImage = r.CaptureedImage != null && r.CaptureedImage.Length > 0 }) .ToList(); return Ok(new ApiResponse { Success = true, Data = dtos }); } catch (Exception ex) { return InternalServerError(ex); } } /// /// GET: /api/attendance/summary/{userId}/{date} /// [HttpGet] [Route("summary/{userId}/{date}")] public async Task GetDailySummary(int userId, string date) { if (!DateTime.TryParse(date, out var parsedDate)) return BadRequest("Invalid date format. Use yyyy-MM-dd"); try { var summary = await _attendanceService.GetDailyAttendanceSummaryAsync(userId, parsedDate); if (summary == null) return NotFound(); return Ok(new ApiResponse { Success = true, Data = new { id = summary.Id, userId = summary.UserId, date = summary.Date.ToString("yyyy-MM-dd"), checkInTime = summary.CheckInTime?.ToString("HH:mm:ss"), checkOutTime = summary.CheckOutTime?.ToString("HH:mm:ss"), workDuration = summary.WorkDuration?.ToString(@"hh\:mm\:ss"), workHours = summary.WorkDuration?.TotalHours, status = summary.AttendanceStatus.ToString() } }); } catch (Exception ex) { return InternalServerError(ex); } } /// /// GET: /api/attendance/records/{userId}?startDate=&endDate= /// [HttpGet] [Route("records/{userId}")] public async Task GetUserRecords(int userId, [FromUri] string startDate = null, [FromUri] string endDate = null) { try { var start = string.IsNullOrEmpty(startDate) ? DateTime.UtcNow.AddMonths(-1) : DateTime.Parse(startDate); var end = string.IsNullOrEmpty(endDate) ? DateTime.UtcNow : DateTime.Parse(endDate); var records = await _attendanceService.GetUserAttendanceAsync(userId, start, end); return Ok(new ApiResponse { Success = true, Message = $"Retrieved {records.Count} records", Data = records.Select(r => new { id = r.Id, userId = r.UserId, timestamp = r.Timestamp.ToString("yyyy-MM-dd HH:mm:ss"), status = r.Status.ToString(), verificationType = r.VerificationType.ToString(), verificationScore = r.VerificationScore, deviceSerialNumber = r.DeviceSerialNumber }) }); } catch (FormatException) { return BadRequest("Invalid date format. Use yyyy-MM-dd"); } catch (Exception ex) { return InternalServerError(ex); } } /// /// GET: /api/attendance/report/monthly?year=2024&month=6&department= /// [HttpGet] [Route("report/monthly")] public async Task GetMonthlyReport(int year, int month, [FromUri] string department = null) { try { var report = await _reportService.GetMonthlyReportAsync(year, month, department); return Ok(new ApiResponse { Success = true, Message = $"Monthly report for {year}-{month:D2}", Data = report }); } catch (Exception ex) { return InternalServerError(ex); } } /// /// GET: /api/attendance/report/daily?date=2024-06-15&department= /// [HttpGet] [Route("report/daily")] public async Task GetDailyReport([FromUri] string date, [FromUri] string department = null) { if (!DateTime.TryParse(date, out var parsedDate)) return BadRequest("Invalid date format. Use yyyy-MM-dd"); try { var report = await _reportService.GetDailyReportAsync(parsedDate, department); return Ok(new ApiResponse { Success = true, Message = $"Daily report for {parsedDate:yyyy-MM-dd}", Data = report.Select(r => new { userId = r.UserId, employeeId = r.EmployeeId, name = r.Name, department = r.Department, date = r.Date.ToString("yyyy-MM-dd"), checkInTime = r.CheckInTime?.ToString("HH:mm:ss"), checkOutTime = r.CheckOutTime?.ToString("HH:mm:ss"), workDuration = r.WorkDuration?.ToString(@"hh\:mm\:ss"), status = r.Status.ToString() }) }); } catch (Exception ex) { return InternalServerError(ex); } } /// /// GET: /api/attendance/report/user/{userId}?startDate=&endDate= /// [HttpGet] [Route("report/user/{userId}")] public async Task GetUserReport(int userId, [FromUri] string startDate = null, [FromUri] string endDate = null) { try { var start = string.IsNullOrEmpty(startDate) ? DateTime.UtcNow.AddMonths(-1) : DateTime.Parse(startDate); var end = string.IsNullOrEmpty(endDate) ? DateTime.UtcNow : DateTime.Parse(endDate); var report = await _reportService.GetUserReportAsync(userId, start, end); if (report == null) return NotFound(); return Ok(new ApiResponse { Success = true, Data = new { user = new { userId = report.User.UserId, employeeId = report.User.EmployeeId, name = report.User.Name, department = report.User.Department }, startDate = report.StartDate.ToString("yyyy-MM-dd"), endDate = report.EndDate.ToString("yyyy-MM-dd"), presentDays = report.PresentDays, lateDays = report.LateDays, absentDays = report.AbsentDays, earlyLeaveDays = report.EarlyLeaveDays, halfDays = report.HalfDays, onLeaveDays = report.OnLeaveDays, totalWorkHours = Math.Round(report.TotalWorkHours, 2), summaries = report.Summaries?.Select(s => new { date = s.Date.ToString("yyyy-MM-dd"), checkInTime = s.CheckInTime?.ToString("HH:mm:ss"), checkOutTime = s.CheckOutTime?.ToString("HH:mm:ss"), workDuration = s.WorkDuration?.ToString(@"hh\:mm\:ss"), status = s.AttendanceStatus.ToString() }) } }); } catch (FormatException) { return BadRequest("Invalid date format. Use yyyy-MM-dd"); } catch (Exception ex) { return InternalServerError(ex); } } /// /// GET: /api/attendance/dashboard-stats — KPIs for the live dashboard. /// [HttpGet] [Route("dashboard-stats")] public async Task GetDashboardStats() { try { var stats = await _attendanceService.GetDashboardStatsAsync(); return Ok(new ApiResponse { Success = true, Data = stats }); } catch (Exception ex) { return InternalServerError(ex); } } } }
