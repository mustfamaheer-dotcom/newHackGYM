using System; using System.Collections.Generic; using System.Linq; using System.Threading.Tasks; using AttendanceSystem.Hubs; using AttendanceSystem.Models; using AttendanceSystem.Repositories; using Microsoft.AspNet.SignalR; namespace AttendanceSystem.Services { /// /// Business rules for attendance events. /// Handles real-time recording, device sync, and daily summary rollups. /// Uses UnitOfWork so cross-repository changes commit atomically. /// public interface IAttendanceService { Task RecordAttendanceAsync(int userId, CheckinStatus status, byte[] capturedImage, VerificationType verification = VerificationType.FaceRecognition, int score = 0, string deviceSerial = null); Task SyncDeviceRecordsAsync(List records, string deviceSerial); Task GetDailyAttendanceSummaryAsync(int userId, DateTime date); Task> GetRangeAttendanceSummariesAsync(int userId, DateTime start, DateTime end); Task> GetUserAttendanceAsync(int userId, DateTime start, DateTime end); Task UpdateDailyAttendanceSummaryAsync(int userId, DateTime date); Task> GetTodayAttendanceAsync(); Task GetDashboardStatsAsync(); } public class AttendanceService : IAttendanceService { private readonly IUnitOfWork _uow; private readonly IAttendancePolicyService _policy; public AttendanceService(IUnitOfWork uow, IAttendancePolicyService policy) { _uow = uow; _policy = policy; } public async Task RecordAttendanceAsync(int userId, CheckinStatus status, byte[] capturedImage, VerificationType verification = VerificationType.FaceRecognition, int score = 0, string deviceSerial = null) { var user = await _uow.Users.GetByIdAsync(userId); if (user == null) throw new InvalidOperationException($"User with ID {userId} not found"); if (user.Status != UserStatus.Active) throw new InvalidOperationException($"User {user.EmployeeId} is not active"); var now = DateTime.UtcNow; var record = new AttendanceRecord { UserId = userId, Timestamp = now, Status = status, CaptureedImage = capturedImage, DeviceSerialNumber = deviceSerial, VerificationType = verification, VerificationScore = score, IsSynced = true, SyncedDate = now }; await _uow.AttendanceRecords.AddAsync(record); await _uow.SaveChangesAsync(); // Update daily summary await UpdateDailyAttendanceSummaryAsync(userId, now.Date); // Broadcast to dashboard via SignalR var hub = GlobalHost.ConnectionManager.GetHubContext(); hub?.Clients.All.attendanceRecorded(new { record.Id, record.UserId, EmployeeId = user.EmployeeId, UserName = user.Name, Timestamp = record.Timestamp, Status = record.Status.ToString(), VerificationType = record.VerificationType.ToString(), record.VerificationScore }); return record; } public async Task SyncDeviceRecordsAsync(List records, string deviceSerial) { var result = new SyncResult(); if (records == null || records.Count == 0) return result; await _uow.BeginTransactionAsync(); try { foreach (var dto in records) { // Skip records for unknown users (FK constraint) var user = await _uow.Users.GetByIdAsync(dto.UserId); if (user == null) { result.Skipped++; continue; } // De-duplicate by (userId, timestamp, device) if (await _uow.AttendanceRecords.ExistsAsync(dto.UserId, dto.Timestamp, dto.DeviceSerialNumber ?? deviceSerial)) { result.Skipped++; continue; } var record = new AttendanceRecord { UserId = dto.UserId, Timestamp = dto.Timestamp, Status = (CheckinStatus)dto.Status, DeviceSerialNumber = dto.DeviceSerialNumber ?? deviceSerial, VerificationType = (VerificationType)dto.VerificationType, VerificationScore = dto.VerificationScore, CaptureedImage = dto.CapturedImage, IsSynced = true, SyncedDate = DateTime.UtcNow }; await _uow.AttendanceRecords.AddAsync(record); result.Inserted++; // Rollup daily summary for each user/date touched await UpdateDailyAttendanceSummaryAsync(dto.UserId, dto.Timestamp.Date); } await _uow.SaveChangesAsync(); await _uow.CommitAsync(); // Update device sync timestamp if (!string.IsNullOrEmpty(deviceSerial)) { await _uow.Devices.UpdateSyncStatusAsync(deviceSerial, DeviceStatus.Online); } } catch { await _uow.RollbackAsync(); throw; } // Broadcast new records to dashboard if (result.Inserted > 0) { var hub = GlobalHost.ConnectionManager.GetHubContext(); hub?.Clients.All.recordsSynced(new { deviceSerial, result.Inserted, result.Skipped }); } return result; } public async Task GetDailyAttendanceSummaryAsync(int userId, DateTime date) => await _uow.AttendanceSummaries.GetByUserAndDateAsync(userId, date); public async Task> GetRangeAttendanceSummariesAsync(int userId, DateTime start, DateTime end) => await _uow.AttendanceSummaries.GetByUserAndDateRangeAsync(userId, start, end); public async Task> GetUserAttendanceAsync(int userId, DateTime start, DateTime end) => await _uow.AttendanceRecords.GetByUserAndDateRangeAsync(userId, start, end); public async Task UpdateDailyAttendanceSummaryAsync(int userId, DateTime date) { var user = await _uow.Users.GetByIdAsync(userId); if (user == null) return false; var summary = await _uow.AttendanceSummaries.GetByUserAndDateAsync(userId, date); var records = await _uow.AttendanceRecords.GetByUserAndDateRangeAsync(userId, date.Date, date.Date.AddDays(1).AddTicks(-1)); if (records.Count == 0) { if (summary != null) { summary.AttendanceStatus = AttendanceStatus.Absent; summary.UpdatedDate = DateTime.UtcNow; } await _uow.SaveChangesAsync(); return true; } var ordered = records.OrderBy(r => r.Timestamp).ToList(); var checkInRecord = ordered.FirstOrDefault(r => r.Status == CheckinStatus.CheckIn); var checkOutRecord = ordered.LastOrDefault(r => r.Status == CheckinStatus.CheckOut); DateTime? checkInTime = checkInRecord?.Timestamp; DateTime? checkOutTime = checkOutRecord?.Timestamp; var status = _policy.DetermineStatus(date, checkInTime, checkOutTime); TimeSpan? workDuration = null; if (checkInTime.HasValue && checkOutTime.HasValue) { workDuration = checkOutTime.Value - checkInTime.Value; if (workDuration.Value.TotalSeconds > GetTodayAttendanceAsync() { var today = DateTime.UtcNow.Date; return await _uow.AttendanceRecords.GetByDateRangeAsync(today, today.AddDays(1).AddTicks(-1)); } public async Task GetDashboardStatsAsync() { var today = DateTime.UtcNow.Date; var activeUsers = await _uow.Users.GetActiveUsersAsync(); var todayRecords = await _uow.AttendanceRecords.GetByDateRangeAsync(today, today.AddDays(1).AddTicks(-1)); var todayCheckedInUserIds = todayRecords .Where(r => r.Status == CheckinStatus.CheckIn) .Select(r => r.UserId) .Distinct() .ToHashSet(); var todaySummaries = await _uow.AttendanceSummaries.GetByDateRangeAsync(today, today); return new DashboardStatsDto { TotalActiveUsers = activeUsers.Count, CheckedInToday = todayCheckedInUserIds.Count, AbsentToday = activeUsers.Count - todayCheckedInUserIds.Count, LateToday = todaySummaries.Count(s => s.AttendanceStatus == AttendanceStatus.Late), OnLeaveToday = todaySummaries.Count(s => s.AttendanceStatus == AttendanceStatus.OnLeave), TotalRecordsToday = todayRecords.Count, DevicesOnline = (await _uow.Devices.GetActiveDevicesAsync()).Count }; } } // ---------- Helper types ---------- public class SyncRecordInput { public int UserId { get; set; } public DateTime Timestamp { get; set; } public int Status { get; set; } public string DeviceSerialNumber { get; set; } public int VerificationType { get; set; } public int VerificationScore { get; set; } public byte[] CapturedImage { get; set; } } public class SyncResult { public int Inserted { get; set; } public int Skipped { get; set; } public int Total => Inserted + Skipped; } public class DashboardStatsDto { public int TotalActiveUsers { get; set; } public int CheckedInToday { get; set; } public int AbsentToday { get; set; } public int LateToday { get; set; } public int OnLeaveToday { get; set; } public int TotalRecordsToday { get; set; } public int DevicesOnline { get; set; } public DateTime GeneratedAt { get; set; } = DateTime.UtcNow; } }
